Polymorphism

Code:
class Dog:

    def speak(self):
        print("Dog barking.")

    def spot(self):
        print("Dog have spots.")


class Cat:
    def speak(self):
        print("Cat barking.")

    def spot(self):
        print("Cat have spots.")


d = Dog()
c = Cat()

d.speak()
c.speak()

d.spot()
c.spot()


# polyphormic way
print("\nusing poly. looping way")
for animal in [d, c]:
    animal.speak()
    animal.spot()


# using polyphormic function
def species(s):
    print("\nUsing poly. func")
    s.speak()
    s.spot()


species(d)
species(c)
